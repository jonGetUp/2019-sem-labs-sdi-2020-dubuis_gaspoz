Process statement

Found in

Examples 

    mux: process(sel, x0, x1)
begin
  if sel = '0' then
    y <= x0;
  elsif sel = '1' then
    y <= x1;
  else
    y <= 'X';
  end if;
end process mux;
      
    count: process(reset, clock)
begin
  if reset = '1' then
    counter <= (others => '0');
  elsif rising_edge(clock) then
    counter <= counter + 1;
  end if;
end process count;
