Examples 

    case sel is
  when "00" => y <= x0;
  when "01" => y <= x1;
  when "10" => y <= x2;
  when "11" => y <= x3;
  when others => null;
end case;
      
    case opCode is
  when add => y1 <= x0;
              y2 <= x1;
  when sub => y1 <= x1;
              y2 <= x0;
  when others => null;
end case;
      
    case value is
  when 1       => nBits <= 1;
  when 2|3     => nBits <= 2;
  when 4 to 7  => nBits <= 3;
  when 8 to 15 => nBits <= 4;
  when others  => nBits <= 0;
end case;
